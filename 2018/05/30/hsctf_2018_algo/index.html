End

A simple dynamic programming problem.

We are asked to find the number of ways to go from the lower left corner to the upper right corner in a map and can only move right or up. Thus, we use two loops, one iterating from the first column to the last and the other iterating from the last row to the first row.

For any position in the map, the number of ways to arrive at that location is 0 if the number at that position is 1; otherwise, it is equal to the number of ways to arrive at the position below it plus the number of ways to arrive at the position left to it.

Let f(i, j)be the ways to arrive at position (i, j), then we have the following equation.

f(i, j) = f(i + 1, j) + f(i, j - 1)

f(N - 1, 0)is initialized as 1 and the answer will be f(0, M - 1).

Note that the edges are initialized in a special way.

This is the code we used.

f = open('tiles_c22a1fe073b3047380c9e44d64bb0dfd706cf259d5c827b799de20a2e9c1d6aa.txt', 'r')

Deadfish

This problem can also be solved using dynamic programming, but it is more complicated than the last one. It also involved the use of breadth first search. Because C++ runs more efficiently than Python, we used C++ to solve this problem.

So we are given this interpreter and asked how to produce a certain out with the shortest input.

def interpret(program):
    m = 257
    p = 3
    out = ""
    program = program.lower()
    n = 0
    for c in program:
        if c == 'i':
            n = (n + 1) % m
        elif c == 'd':
            n = (n - 1) % m
        elif c == 'p':
            n = n ** p % m
        elif c == 'o':
            out = out + str(n)
        else:
            return None
    return out

Because the longest n can at most be three digits long, so each time an o command is used, three characters at most can be appended to the string out.

For efficiency, we pre-calculate the number of steps required to change n from i to j (using commands i, d, and/or p) for each (i, j) pair in range 0 to 256 using breadth-first search.

For each i, we create a value pair consisting of (num, step) indicating the number of steps required to transform i into num and appending it to out.

Also, note that 0 can still be decreased by 1, and then -1 % 257 will be equal to 256.

const int mod = 257, power = 3, inf = 1 << 30;
int cost[300][300], f[4][257];

struct value {
    int num, step;
};

int command(int n, string c) {

    if (c == "i") {
        return (n + 1) % mod;
    } else if (c == "d") {
        return (((n - 1) % mod) + mod) % mod;
    } else if (c == "p") {
        return (n * n * n) % mod;
    } else {
        return -1;
    }

}

void pre_calculate() {

    for (int i = 0; i < 257; i += 1) {

        cost[i][i] = 1;

        value first = {i, 1}; // 1 step is needed for appending characters to out
        queue<value> q;

        q.push(first);

        while (!q.empty()) {

            value curr = q.front();
            q.pop();

            int v = curr.num, s = curr.step;

            if (cost[i][v] == 0 or s <= cost[i][v]) {
                cost[i][v] = s;
            } else {
                continue;
            }

            bool flag = true;
            for (int j = 0; j < 257; j += 1) {
                if (i == j) {
                    continue;
                }
                if (cost[i][j] == 0) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                break;
            }

            q.push({command(v, "i"), s + 1});
            q.push({command(v, "d"), s + 1});
            q.push({command(v, "p"), s + 1});

        }

    }
}

After pre-calculating the 2-dimensional array cost, we can move on to the dynamic programming part, which is where things get more interesting.

In this problem, we use f(i, j) to represent the length of the shortest input required to append the first i - 1 characters to out and end up with the value of n being j after doing so.

For instance, if we want to obtain the output "654321", f(2, 5) would represent the situation where out = "65" and n = 5.

Due to the fact that n can be at most 3 digits long, f(i, j) will only depend on the values of f(i - 1, k), f(i - 2, k), and f(i - 2, k) for k in [0, 256].

f(i, j) = min(f(i, j), f(i - 1, k) + cost(k, v), f(i - 2, k) + cost(k, v), f(i - 3, k) + cost(k, v)) for k in [0, 256] and v is equal to the portion of out between indices i - 1 and i, i - 2 and i, i - 3 and i respectively.

Also, to save memory, and because one state is only affected by the three states before it, we used a technique to reduce the size of the array used for dynamic programming.

for (int i = 0; i < 4; i += 1) {
    for (int j = 0; j < 257; j += 1) {
        f[i][j] = inf;
    }
}

f[0][0] = 0;  // when out is empty and n is initially 0

for (int i = 1; i <= target.length(); i += 1) {

    for (int j = 0; j < 257; j += 1) {
        f[i % 4][j] = inf;
    }

    for (int j = 0; j < 257; j += 1) {

        int v;

        v = char2int(target.substr(i - 1, 1));
        if (f[i % 4][v] > f[(i + 4 - 1) % 4][j] + cost[j][v]) {
            f[i % 4][v] = f[(i + 4 - 1) % 4][j] + cost[j][v];
        }

        if (i > 1) {
            v = char2int(target.substr(i - 2, 2));
            if (f[i % 4][v] > f[(i + 4 - 2) % 4][j] + cost[j][v]) {
                f[i % 4][v] = f[(i + 4 - 2) % 4][j] + cost[j][v];
            }
        }

        if (i > 2) {
            v = char2int(target.substr(i - 3, 3));
            if (v < 257) {
                if (f[i % 4][v] > f[(i + 4 - 3) % 4][j] + cost[j][v]) {
                    f[i % 4][v] = f[(i + 4 - 3) % 4][j] + cost[j][v];
                }
            }
        }

    }
}

The answer will be the minimum in f(N, k) for k in [0, 256] where N is the length of the desired out.

The entire program is as follows.

#include <iostream>
#include <queue>
#include <cmath>
#include <string>
#include <fstream>

using namespace std;

struct value {
    int num, step;
};

const int mod = 257, power = 3, inf = 1 << 30;
int cost[300][300], f[4][257];
string target;

ifstream fin("input.txt");
ofstream fout("ans.txt");

int command(int n, string c) {

    if (c == "i") {
        return (n + 1) % mod;
    } else if (c == "d") {
        return (((n - 1) % mod) + mod) % mod;
    } else if (c == "p") {
        return (n * n * n) % mod;
    } else {
        return -1;
    }

}

int char2int(string s) {

    int res = 0;

    for (int i = 0; i < s.length(); i += 1) {
        res *= 10;
        res += (int)(s[i] - '0');
    }

    return res;

}

void pre_calculate() {

    for (int i = 0; i < 257; i += 1) {

        cost[i][i] = 1;

        value first = {i, 1};
        queue<value> q;

        q.push(first);

        while (!q.empty()) {

            value curr = q.front();
            q.pop();

            int v = curr.num, s = curr.step;

            if (cost[i][v] == 0 or s <= cost[i][v]) {
                cost[i][v] = s;
            } else {
                continue;
            }

            bool flag = true;
            for (int j = 0; j < 257; j += 1) {
                if (i == j) {
                    continue;
                }
                if (cost[i][j] == 0) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                break;
            }

            q.push({command(v, "i"), s + 1});
            q.push({command(v, "d"), s + 1});
            q.push({command(v, "p"), s + 1});

        }

    }
}

int main() {

    pre_calculate();

    fin >> target;

    for (int i = 0; i < 4; i += 1) {
        for (int j = 0; j < 257; j += 1) {
            f[i][j] = inf;
        }
    }

    f[0][0] = 0;

    for (int i = 1; i <= target.length(); i += 1) {

        for (int j = 0; j < 257; j += 1) {
            f[i % 4][j] = inf;
        }

        for (int j = 0; j < 257; j += 1) {

            int v;

            v = char2int(target.substr(i - 1, 1));
            if (f[i % 4][v] > f[(i + 4 - 1) % 4][j] + cost[j][v]) {
                f[i % 4][v] = f[(i + 4 - 1) % 4][j] + cost[j][v];
            }

            if (i > 1) {
                v = char2int(target.substr(i - 2, 2));
                if (f[i % 4][v] > f[(i + 4 - 2) % 4][j] + cost[j][v]) {
                    f[i % 4][v] = f[(i + 4 - 2) % 4][j] + cost[j][v];
                }
            }

            if (i > 2) {
                v = char2int(target.substr(i - 3, 3));
                if (v < 257) {
                    if (f[i % 4][v] > f[(i + 4 - 3) % 4][j] + cost[j][v]) {
                        f[i % 4][v] = f[(i + 4 - 3) % 4][j] + cost[j][v];
                    }
                }
            }
        }
    }

    int ans = inf;

    for (int j = 0; j < 257; j += 1) {
        if (f[target.length() % 4][j] < ans) {
            ans = f[target.length() % 4][j];
        }
    }

    fout << ans << endl;

    return 0;

}
